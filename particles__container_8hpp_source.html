270 KOKKOS_INLINE_FUNCTION void
271 operator()(const int i, std::size_t& update, const bool final) const
272 {
273
280
281 // Check if the particle is non-idle.
282 const bool is_inactive = (status(i) != MC::Status::Idle);
283
284 // Capture the current prefix sum of inactive particles.
285 const std::size_t scan_index = update;
286
287 // Increment the shared counter if the particle is inactive.
288 update += is_inactive ? 1 : 0;
289
290 // Final pass: Compact the container by replacing inactive particles.
291 if (final && is_inactive && scan_index < to_remove)
292 {
293
294 const auto inactive_slot = i; // Index of the "gap" to fill
295
296 // Atomically find a replacement particle from the end of the
297 // container. The replacement must be alive (idle) and not the same as
298 // the particle being removed.
299 // Index of the particle that will "fill" the inactive slot
300 auto replacement_index =
301 last_used_index - Kokkos::atomic_fetch_add(&offset(), 1);
302 while (status(replacement_index) != MC::Status::Idle ||
303 replacement_index == static_cast<std::size_t>(inactive_slot))
304 {
305 replacement_index =
306 last_used_index - Kokkos::atomic_fetch_add(&offset(), 1);
307 }
308
309 // Mark the removed particle's position as idle.
310 status(inactive_slot) = MC::Status::Idle;
311
312 // Merge position EZ
313 // No need atomic: final executed exactly once
314 Kokkos::atomic_exchange(&position(inactive_slot),
315 position(replacement_index));
316
317 // TODO Use hierachical parallism here, thread range is likely to work
318 for (std::size_t i_properties = 0; i_properties < M::n_var;
319 ++i_properties)
320 {
321 model(inactive_slot, i_properties) =
322 model(replacement_index, i_properties);
323 }
324 ages(inactive_slot, 0) = ages(replacement_index, 0);
325 ages(inactive_slot, 1) = ages(replacement_index, 1);
326 }
327 }
328
329 MC::ParticleStatus status;
330 M::SelfParticle model;
331 MC::ParticlePositions position;
332 MC::ParticleAges ages;
333 Kokkos::View<std::size_t, ComputeSpace> offset;
334 std::size_t to_remove;
335 std::size_t last_used_index;
336 };

504 const MC::KPRNG::pool_type& random_pool, std::size_t idx1) const
505 {
506 if (Kokkos::atomic_load(&buffer_index()) < buffer_model.extent(0))
507 {

509 Model::division(random_pool, idx1, idx2, model, buffer_model);
510 buffer_position(idx2) = position(idx1);
511 ages(idx1, 1) = 0;
512 return true;
513 }

517 template <ModelType Model> void ParticlesContainer<Model>::merge_buffer()
518 {
519 PROFILE_SECTION("ParticlesContainer::merge_buffer")
520 const auto original_size = n_used_elements;
521 const auto n_add_item = buffer_index();
522 if (n_add_item == 0)
523 {
524 return;
525 }
526 _resize(original_size + n_add_item);
527 Kokkos::parallel_for("insert_merge",
528 TeamPolicy(n_add_item, Kokkos::AUTO, Model::n_var),
529 InsertFunctor<Model>(original_size,
530 model,
531 position,
532 ages,
533 buffer_model,
534 buffer_position));
535
536 buffer_index() = 0;
537 n_used_elements += n_add_item;
538 __allocate_buffer__();
539 }

541 template <ModelType Model>
542 void ParticlesContainer<Model>::_resize(std::size_t new_size, bool force)
543 {
544 PROFILE_SECTION("ParticlesContainer::_resize")
545
546 // Ensure new_size is greater than zero
547 if (new_size > 0)
548 {
549 // Determine if resizing is necessary based on the condition
550 if (new_size > n_allocated_elements || force)
551 {
552 // Calculate the new allocated size
553 const auto new_allocated_size = static_cast<std::size_t>(
554 std::ceil(static_cast<double>(new_size) * allocation_factor));
555
556 // Update the allocated size
557 n_allocated_elements = new_allocated_size;
558
559 // Perform the resizing on all relevant data containers
560 Kokkos::resize(position, n_allocated_elements);
561 Kokkos::resize(model,
562 n_allocated_elements,
563 Model::n_var); // use 2nd dim resize if dynamic
564 Kokkos::resize(status, n_allocated_elements);
565 Kokkos::resize(ages, n_allocated_elements);
566
567 // Handle resizing for weights based on model type
568 if constexpr (ConstWeightModelType<Model>)
569 {
570 Kokkos::resize(weights, 1); // Fixed size for ConstWeightModelType
571 }
572 else
573 {
574 Kokkos::resize(weights, n_allocated_elements);
575 }
576 }
577 }

580 template <ModelType Model>
581 void ParticlesContainer<Model>::__allocate_buffer__()
582 {
583 PROFILE_SECTION("ParticlesContainer::__allocate_buffer__")
584 auto buffer_size = buffer_position.extent(0);
585 if (static_cast<double>(buffer_size) /
586 static_cast<double>(n_allocated_elements) <
587 buffer_ratio)
588 {
589 buffer_size = static_cast<std::size_t>(
590 std::ceil(static_cast<double>(n_allocated_elements) * buffer_ratio));
591
592 // Realloc because not needed to keep buffer as it has been copied
593 Kokkos::realloc(buffer_position, buffer_size);
594 Kokkos::realloc(buffer_model,
595 buffer_size,
596 Model::n_var); // use 2nd dim resize if dynamic
597 buffer_index() = 0;
598 }

640 template <ModelType M>
641 void ParticlesContainer<M>::remove_inactive_particles(std::size_t to_remove)
642 {
643
644 PROFILE_SECTION("ParticlesContainer::remove_inactive_particles")
645 if (to_remove == 0)
646 {
647 return;
648 }
649
650 if (to_remove == n_used_elements)
651 {
652 _resize(0, true);
653 n_used_elements = 0;
654 inactive_counter = 0;
655 }
656 else if (to_remove > n_used_elements)
657 {
658 throw std::runtime_error(
659 "remove_inactive_particles: Error in kernel cannot remove more "
660 "element than existing");
661 }
662 else
663 {
664
665 const auto new_used_item = n_used_elements - to_remove;
666
667 const auto last_used_index = n_used_elements - 1;
668 Kokkos::parallel_scan(
669 "find_and_fill_gap",
670 Kokkos::RangePolicy<ComputeSpace>(0, n_used_elements),
671 CompactParticlesFunctor<M>(
672 status, model, position, ages, to_remove, last_used_index));
673
674 Kokkos::fence();
675 KOKKOS_ASSERT(this->position.extent(0) == n_allocated_elements);
676 KOKKOS_ASSERT(this->model.extent(0) == n_allocated_elements);
677 KOKKOS_ASSERT(this->status.extent(0) == n_allocated_elements);
678 n_used_elements = new_used_item;
679 if (static_cast<double>(n_used_elements) /
680 static_cast<double>(n_allocated_elements) <=
681 AutoGenerated::MC::shink_ratio)
682 {
683 // std::cout << "SHRINK" << std::endl;
684 _resize(n_used_elements * 2, false);
685 }
686 inactive_counter = inactive_counter - to_remove;
687 };
