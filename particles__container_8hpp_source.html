149 FillGapFunctor(MC::ParticleStatus _status,
150 M::SelfParticle _model,
151 MC::ParticlePositions _position,
152 MC::ParticleAges _ages,
153 std::size_t _to_remove,
154 std::size_t _last_used_index)
155 : status(std::move(_status)), model(std::move(_model)),
156 position(std::move(_position)), ages(std::move(_ages)),

160

163
164 KOKKOS_INLINE_FUNCTION void
165 operator()(const int i, std::size_t& update, const bool final) const
166 {
167 const bool is_dead = (status(i) != MC::Status::Idle);
168
169 const std::size_t scan_index = update;
170 update += is_dead ? 1 : 0;
171
172 if (final && is_dead && scan_index < to_remove)
173 {
174
175 const auto i_to_remove = i;
176 auto idx_to_move =
177 last_used_index - Kokkos::atomic_fetch_add(&offset(), 1);
178 while (status(idx_to_move) != MC::Status::Idle ||
179 idx_to_move == static_cast<std::size_t>(i_to_remove))
180 {
181 idx_to_move =
182 last_used_index - Kokkos::atomic_fetch_add(&offset(), 1);
183 }
184
185 // Merge position EZ
186 status(i_to_remove) = MC::Status::Idle;
187
188 // No need atomic: final executed exactly once
189 Kokkos::atomic_exchange(&position(i_to_remove),
190 position(idx_to_move));
191
192 // TODO Use hierachical parallism here, thread range is likely to work
193 for (std::size_t i_properties = 0; i_properties < M::n_var;
194 ++i_properties)
195 {
196 model(i_to_remove, i_properties) = model(idx_to_move, i_properties);
197 }
198 ages(i_to_remove, 0) = ages(idx_to_move, 0);
199 ages(i_to_remove, 1) = ages(idx_to_move, 1);
200 }
201 }

203 MC::ParticleStatus status;
204 M::SelfParticle model;
205 MC::ParticlePositions position;
206 MC::ParticleAges ages;
207 Kokkos::View<std::size_t, ComputeSpace> offset;
208 std::size_t to_remove;
209 std::size_t last_used_index;

361 Model::division(random_pool, idx1, idx2, model, buffer_model);

369 template <ModelType Model> void ParticlesContainer<Model>::merge_buffer()
370 {
371 PROFILE_SECTION("ParticlesContainer::merge_buffer")
372 const auto original_size = n_used_elements;
373 const auto n_add_item = buffer_index();
374 if (n_add_item == 0)
375 {
376 return;
377 }
378 _resize(original_size + n_add_item);
379 Kokkos::parallel_for("insert_merge",
380 TeamPolicy(n_add_item, Kokkos::AUTO, Model::n_var),
381 InsertFunctor<Model>(original_size,
382 model,
383 position,
384 ages,
385 buffer_model,
386 buffer_position));
387
388 buffer_index() = 0;
389 n_used_elements += n_add_item;
390 __allocate_buffer__();
391 }

393 template <ModelType Model>
394 void ParticlesContainer<Model>::_resize(std::size_t new_size, bool force)
395 {
396 PROFILE_SECTION("ParticlesContainer::_resize")
397
398 // Ensure new_size is greater than zero
399 if (new_size > 0)
400 {
401 // Determine if resizing is necessary based on the condition
402 if (new_size > n_allocated_elements || force)
403 {
404 // Calculate the new allocated size
405 const auto new_allocated_size = static_cast<std::size_t>(
406 std::ceil(static_cast<double>(new_size) * allocation_factor));
407
408 // Update the allocated size
409 n_allocated_elements = new_allocated_size;
410
411 // Perform the resizing on all relevant data containers
412 Kokkos::resize(position, n_allocated_elements);
413 Kokkos::resize(model,
414 n_allocated_elements,
415 Model::n_var); // use 2nd dim resize if dynamic
416 Kokkos::resize(status, n_allocated_elements);
417 Kokkos::resize(ages, n_allocated_elements);
418
419 // Handle resizing for weights based on model type
420 if constexpr (ConstWeightModelType<Model>)
421 {
422 Kokkos::resize(weights, 1); // Fixed size for ConstWeightModelType
423 }
424 else
425 {
426 Kokkos::resize(weights, n_allocated_elements);
427 }
428 }
429 }

432 template <ModelType Model>
433 void ParticlesContainer<Model>::__allocate_buffer__()
434 {
435 PROFILE_SECTION("ParticlesContainer::__allocate_buffer__")
436 auto buffer_size = buffer_position.extent(0);
437 if (static_cast<double>(buffer_size) /
438 static_cast<double>(n_allocated_elements) <
439 buffer_ratio)
440 {
441 buffer_size = static_cast<std::size_t>(
442 std::ceil(static_cast<double>(n_allocated_elements) * buffer_ratio));
443
444 // Realloc because not needed to keep buffer as it has been copied
445 Kokkos::realloc(buffer_position, buffer_size);
446 Kokkos::realloc(buffer_model,
447 buffer_size,
448 Model::n_var); // use 2nd dim resize if dynamic
449 buffer_index() = 0;
450 }

491 template <ModelType M>
492 void ParticlesContainer<M>::clean_dead(std::size_t to_remove)
493 {
494 PROFILE_SECTION("ParticlesContainer::remove_dead")
495 if (to_remove == 0)
496 {
497 return;
498 }
499
500 if (to_remove == n_used_elements)
501 {
502 _resize(0, true);
503 n_used_elements = 0;
504 }
505 else if (to_remove > n_used_elements)
506 {
507 throw std::runtime_error("clean_dead: Error in kernel cannot remove more "
508 "element than existing");
509 }
510 else
511 {
512
513 const auto new_used_item = n_used_elements - to_remove;
514
515 const auto last_used_index = n_used_elements - 1;
516 Kokkos::parallel_scan(
517 "find_and_fill_gap",
518 Kokkos::RangePolicy<ComputeSpace>(0, n_used_elements),
519 FillGapFunctor<M>(
520 status, model, position, ages, to_remove, last_used_index));
521
522 Kokkos::fence();
523 KOKKOS_ASSERT(this->position.extent(0) == n_allocated_elements);
524 KOKKOS_ASSERT(this->model.extent(0) == n_allocated_elements);
525 KOKKOS_ASSERT(this->status.extent(0) == n_allocated_elements);
526 n_used_elements = new_used_item;
527 if (static_cast<double>(n_used_elements) /
528 static_cast<double>(n_allocated_elements) <=
529 0.1)
530 {
531 // std::cout << "SHRINK" << std::endl;
532 _resize(n_used_elements * 2, false);
533 }
534 };
