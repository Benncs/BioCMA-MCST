174 KOKKOS_INLINE_FUNCTION void
175 TwoMetaNb::init([[maybe_unused]] const MC::KPRNG::pool_type& random_pool,
176 std::size_t idx,
177 const SelfParticle& arr)
178 {
179
180 // auto& v = init_uptake_cst;
181 constexpr auto local_ac = adder_dist;
182
183 constexpr auto length_dist = MC::Distributions::TruncatedNormal<FloatType>(
184 l_c_m / 2, l_c_m / 5., l_min_m, l_max_m);
185
186 constexpr auto mu_nu_dist = nu_max_kg_s * 0.1;
187 constexpr auto nu_1_initial_dist =
188 MC::Distributions::TruncatedNormal<FloatType>(
189 mu_nu_dist, mu_nu_dist / 7., 0., static_cast<double>(nu_max_kg_s));
190
191 auto gen = random_pool.get_state();
192 auto l = length_dist.draw(gen);
193
194 GET_PROPERTY(Self::particle_var::length) = l;
195 GET_PROPERTY(Self::particle_var::l_cp) = l + local_ac.draw(gen);
196 GET_PROPERTY(particle_var::nu1) = nu_1_initial_dist.draw(gen);
197 random_pool.free_state(gen);
198 GET_PROPERTY(particle_var::contrib_phi_s) = 0;
199
200 Uptake<Self>::init(random_pool, idx, arr);
201 }

203 KOKKOS_INLINE_FUNCTION MC::Status
204 TwoMetaNb::update([[maybe_unused]] const MC::KPRNG::pool_type& random_pool,
205 FloatType d_t,
206 std::size_t idx,
207 const SelfParticle& arr,
208 const MC::LocalConcentration& concentrations)
209 {
210 const auto phi_s =
211 Uptake<Self>::uptake_step(phi_max,
212 d_t,
213 idx,
214 arr,
215 concentrations,
216 &GET_PROPERTY(Self::particle_var::phi_pts));
217
218 const auto o = Kokkos::max(static_cast<FloatType>(concentrations(1)), 0.F);
219
220 const auto phi_o2 = (phi_o2_max)*o / (o + k_o); // gO2/s
221
222 const auto nu_1_star =
223 y_sx_1 * MolarMassG *
224 Kokkos::min(phi_s / MolarMassG,
225 phi_o2 / MolarMassO2 / y_os_molar); // gX/s
226
227 const auto s_1_star = (1.F / y_sx_1 * nu_1_star);
228
229 const auto phi_s_residual_1_star = Kokkos::max(phi_s - s_1_star, 0.F);
230 KOKKOS_ASSERT(phi_s_residual_1_star >= 0.F);
231
232 const auto nu_2_star = y_sx_2 * phi_s_residual_1_star; // gX/s
233
234 auto& nu_eff_1 = GET_PROPERTY(Self::particle_var::nu_eff_1);
235 auto& nu_eff_2 = GET_PROPERTY(Self::particle_var::nu_eff_2);
236 auto& nu_1 = GET_PROPERTY(Self::particle_var::nu1);
237 auto& nu_2 = GET_PROPERTY(Self::particle_var::nu2);
238
239 nu_eff_1 = Kokkos::min(nu_1_star, nu_1); // gX/s
240
241 const auto s_1 = (1.F / y_sx_1 * nu_eff_1);
242 const auto phi_s_residual_1 = Kokkos::max(phi_s - s_1, 0.F);
243
244 nu_eff_2 = Kokkos::min(y_sx_2 * phi_s_residual_1,
245 nu_2); // gX/s
246
247 const auto sum_nu = (nu_eff_1 + nu_eff_2);
248
249 const auto s_growth = s_1 + (1 / y_sx_2 * nu_eff_2);
250
251 const auto s_overflow = phi_s - s_growth;
252
253 KOKKOS_ASSERT(nu_eff_1 >= 0.F);
254 KOKKOS_ASSERT(nu_eff_2 >= 0.F);
255
256 // CONTRIBS
257 GET_PROPERTY(Self::particle_var::contrib_phi_s) = -phi_s;
258 GET_PROPERTY(Self::particle_var::contrib_phi_o2) =
259 -1.F *
260 ((1.F / y_sx_1 / MolarMassG * y_os_molar * MolarMassO2 * nu_eff_1) +
261 0.F * nu_eff_2);
262
263 GET_PROPERTY(Self::particle_var::contrib_phi_ac) =
264 nu_eff_2 / y_sx_2 * y_sa + (s_overflow > 0. ? y_sa * (s_overflow) : 0);
265
266 // ODE
267 nu_1 += d_t * ((nu_1_star - nu_1) / tau_1);
268
269 nu_2 += d_t * ((nu_2_star - nu_2) / tau_2);
270
271 GET_PROPERTY(Self::particle_var::length) += d_t * (sum_nu / lin_density);
272
273 GET_PROPERTY(Self::particle_var::age) += d_t;
274
275 return (GET_PROPERTY(Self::particle_var::length) >
276 GET_PROPERTY(Self::particle_var::l_cp))
277 ? MC::Status::Division
278 : MC::Status::Idle;
279 }

281 KOKKOS_INLINE_FUNCTION void
282 TwoMetaNb::division(const MC::KPRNG::pool_type& random_pool,
283 std::size_t idx,
284 std::size_t idx2,
285 const SelfParticle& arr,
286 const SelfParticle& child_buffer_arr)
287 {
288 constexpr FloatType half = 0.5;
289 constexpr auto local_ac = adder_dist;
290 const FloatType new_current_length =
291 GET_PROPERTY(particle_var::length) / static_cast<FloatType>(2.);
292
293 GET_PROPERTY(Self::particle_var::length) = new_current_length;
294 GET_PROPERTY(Self::particle_var::age) = 0;
295
296 GET_PROPERTY_FROM(idx2, child_buffer_arr, Self::particle_var::length) =
297 new_current_length;
298
299 GET_PROPERTY_FROM(idx2, child_buffer_arr, Self::particle_var::age) = 0;
300
301 const auto nu_1_o =
302 GET_PROPERTY_FROM(idx, arr, Self::particle_var::nu_eff_1);
303 const auto nu_2_o =
304 GET_PROPERTY_FROM(idx, arr, Self::particle_var::nu_eff_2);
305
306 auto gen = random_pool.get_state();
307
308 GET_PROPERTY(Self::particle_var::l_cp) =
309 new_current_length + local_ac.draw(gen);
310
311 if (nu_1_o != 0)
312 {
313 GET_PROPERTY_FROM(idx2, child_buffer_arr, Self::particle_var::nu1) =
314 MC::Distributions::TruncatedNormal<FloatType>::draw_from(
315 gen, nu_1_o, nu_1_o * half, 0.F, 1.F);
316 }
317
318 if (nu_2_o != 0)
319 {
320 GET_PROPERTY_FROM(idx2, child_buffer_arr, Self::particle_var::nu2) =
321 MC::Distributions::TruncatedNormal<FloatType>::draw_from(
322 gen, nu_2_o, nu_2_o * half, 0.F, 1.F);
323 }
324
325 GET_PROPERTY_FROM(idx2, child_buffer_arr, Self::particle_var::l_cp) =
326 new_current_length + local_ac.draw(gen);
327 random_pool.free_state(gen);
328
329 Uptake<Self>::division(random_pool, idx, idx2, arr, child_buffer_arr);
330 }
