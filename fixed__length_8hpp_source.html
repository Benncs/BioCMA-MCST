124 KOKKOS_INLINE_FUNCTION MC::Status
125 FixedLength::update([[maybe_unused]] const MC::KPRNG::pool_type& random_pool,
126 FloatType d_t,
127 std::size_t idx,
128 const SelfParticle& arr,
129 const MC::LocalConcentration& c)
130 {
131 const auto s = static_cast<FloatType>(c(0));
132 const FloatType g = s / (k + s);
133 const FloatType phi_s = phi_s_max * g;
134 // const FloatType ldot = l_dot_max * static_cast<FloatType>(g > 0.1);
135 // GET_PROPERTY(Self::particle_var::length) += d_t * ldot;
136 //
137 const FloatType ldot = l_dot_max * g;
138 const FloatType d_length = d_t * ldot;
139 GET_PROPERTY(Self::particle_var::length) += d_length / (1.0 + d_t * ldot);
140
141 GET_PROPERTY(Self::particle_var::phi_s) = -phi_s;
142 return check_div(GET_PROPERTY(Self::particle_var::length),
143 GET_PROPERTY(Self::particle_var::l_max));
144 }
