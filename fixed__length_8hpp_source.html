176 KOKKOS_INLINE_FUNCTION MC::Status
177 FixedLength::update([[maybe_unused]] const MC::KPRNG::pool_type& random_pool,
178 FloatType d_t,
179 std::size_t idx,
180 const SelfParticle& arr,
181 const MC::LocalConcentration& c)
182 {
183 const auto s = static_cast<FloatType>(c(0));
184 const FloatType g = s / (k + s);
185 const FloatType phi_s = phi_s_max * g;
186 const FloatType ldot = l_dot_max * g;
187 GET_PROPERTY(Self::particle_var::length) += d_t * ldot;
188 GET_PROPERTY(Self::particle_var::phi_s) = -phi_s;
189 return check_div(GET_PROPERTY(Self::particle_var::length),
190 GET_PROPERTY(Self::particle_var::l_max));
191 }
