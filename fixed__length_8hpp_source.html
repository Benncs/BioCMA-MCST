167 KOKKOS_INLINE_FUNCTION MC::Status
168 FixedLength::update([[maybe_unused]] const MC::KPRNG::pool_type& random_pool,
169 FloatType d_t,
170 std::size_t idx,
171 const SelfParticle& arr,
172 const MC::LocalConcentration& c)
173 {
174 const auto s = static_cast<FloatType>(c(0));
175 const FloatType g = s / (k + s);
176 const FloatType phi_s = phi_s_max * g;
177 const FloatType ldot = l_dot_max * g;
178 GET_PROPERTY(Self::particle_var::length) += d_t * ldot;
179 GET_PROPERTY(Self::particle_var::phi_s) = -phi_s;
180
181 return (GET_PROPERTY(Self::particle_var::length) >=
182 GET_PROPERTY(particle_var::l_max))
183 ? MC::Status::Division
184 : MC::Status::Idle;
185 }
