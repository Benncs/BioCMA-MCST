166 (void)_tag;

168 const std::size_t p0 = team.league_rank() * m_particle_per_team;
169 const auto& c = m_particles.contribs;
170 const auto& status = m_particles.status;
171 const auto n_particle = m_particles.n_particles();
172 const auto& positions = m_particles.position;
173 constexpr std::size_t work_per_thread = 32;
174
175 const auto upper_bound = ((p0 + m_particle_per_team) >= n_particle)
176 ? n_particle - p0
177 : m_particle_per_team;
178 KOKKOS_ASSERT(upper_bound >= 0 && upper_bound < n_particle);
179 const auto range = (upper_bound + work_per_thread - 1) / work_per_thread;
180
181 KOKKOS_ASSERT(m_particle_per_team % work_per_thread == 0);
182
183 Kokkos::parallel_for(
184 Kokkos::TeamThreadRange(team, 0, range),
185 [&](const std::size_t i)
186 {
187 auto access = m_contribution_scatter.access();
188 for (std::size_t k = 0; k < work_per_thread; ++k)
189 {
190 const std::size_t p = p0 + i * work_per_thread + k;
191 if (status(p) != MC::Status::Idle || i >= n_particle)
192 {
193 return;
194 }
195 const double weight = m_particles.get_weight(p);
196 const auto pos = positions(p);
197 Kokkos::parallel_for(Kokkos::ThreadVectorRange(team, 0, M::n_c),
198 [&](const int j)
199 { access(j, pos) += weight * c(p, j); });
