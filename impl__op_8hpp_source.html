354 template <POD_t T>
355 std::optional<std::vector<T>>
356 recv_v(size_t source, MPI_Status* status, size_t tag) noexcept
357 {
358 std::vector<T> buf;
359
360 // Receive the size of the vector
361 auto opt_size = recv<size_t>(source, status, tag);
362 if (!opt_size.has_value())
363 {
364 return std::nullopt; // Return early if size reception fails
365 }
366 size_t buf_size = opt_size.value();
367
368 // Resize the buffer
369 buf.resize(buf_size);
370
371 MPI_Datatype datatype = get_type<T>();
372
373 // Receive the vector data
374 int recv_status = MPI_Recv(buf.data(),
375 static_cast<int>(buf_size),
376 datatype,
377 source,
378 tag,
379 MPI_COMM_WORLD,
380 status);
381
382 if (recv_status != MPI_SUCCESS)
383 {
384 return std::nullopt; // Return early if MPI_Recv fails
385 }
386
387 return buf; // Return the received vector
388 }

434 template <POD_t T>
435 int broadcast(std::vector<T>& data, size_t root, size_t current_rank)
436 {
437
438 size_t data_size = 0;
439 if (current_rank == root)
440 {
441 data_size = data.size();
442 }
443 broadcast(data_size, root);
444 if (current_rank != root)
445 {
446 data.resize(data_size);
447 }
448
449 return MPI_Bcast(data.data(),
450 data_size,
451 get_type<T>(),
452 static_cast<int>(root),
453 MPI_COMM_WORLD);

456 template <POD_t T> int _broadcast_unsafe(T* data, size_t _size, size_t root)
457 {
458
459 if (data == nullptr)
460 {
461 throw std::invalid_argument("Data pointer is null");
462 }
463 if (_size == 0 || _size > std::numeric_limits<size_t>::max())
464 {
465 throw std::invalid_argument("Error size");
466 }
467
468 int comm_size = 0;
469 MPI_Comm_size(MPI_COMM_WORLD, &comm_size);
470 if (root >= static_cast<size_t>(comm_size))
471 {
472 throw std::invalid_argument("Root process rank is out of range");
473 }
474
475 // Broadcast operation
476 return MPI_Bcast(
477 data, _size, get_type<T>(), static_cast<int>(root), MPI_COMM_WORLD);
478 }

582 {
583
584 for (int j = 1; j < static_cast<int>(info.n_rank); ++j)
585 {
586 if (sign != WrapMPI::SIGNALS::NOP)
587 {
588 MPI_Send(&sign, sizeof(sign), MPI_CHAR, j, 0, MPI_COMM_WORLD);

590
591 (
