87 KOKKOS_INLINE_FUNCTION void
88 SimpleModel::init([[maybe_unused]] const MC::KPRNG::pool_type& random_pool,
89 std::size_t idx,
90 const SelfParticle& arr)
91 {
92 MODEL_CONSTANT auto division_time_d = MC::Distributions::TruncatedNormal<FloatType>(
93 500, 500./2.,10,1200); //-Kokkos::log(random_number) / frequency_division;
94 MODEL_CONSTANT auto l_distribution = MC::Distributions::TruncatedNormal<FloatType>(
95 l_min_m, l_min_m / 5., l_min_m * 0.5, l_max_m);
96 auto gen = random_pool.get_state();
97 arr(idx, static_cast<int>(particle_var::length)) = l_distribution.draw(gen);
98 GET_PROPERTY(SimpleModel::particle_var::t_div) = division_time_d.draw(gen);
99 random_pool.free_state(gen);
100 arr(idx, static_cast<int>(particle_var::age)) = 0;
101 arr(idx, static_cast<int>(particle_var::phi_s)) = 0;
102
103 Uptake<SimpleModel>::init(random_pool, idx, arr);
104 }
