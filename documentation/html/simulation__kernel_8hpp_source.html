37  size_t next = i_neighbor(0); // Default to the first neighbor

40  for (int k_neighbor = 0; k_neighbor < max_neighbor - 1; ++k_neighbor)
41  {
42  // Get the cumulative probability range for the current neighbor
43  const auto pi = cumulative_probability(i_compartment, k_neighbor);
44  const auto pn = cumulative_probability(i_compartment, k_neighbor + 1);
45 
46  // Check if the random number falls within the probability range
47  if (random_number <= pn && pi <= random_number)
48  {
49  next = i_neighbor(k_neighbor + 1); // Update to the next neighbor
50  // No need to break, as we're looking for the last valid neighbor in the
51  // range
52  }
53  }
