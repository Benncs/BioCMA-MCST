173  Kokkos::parallel_for(
174  "process", list.size(), KOKKOS_LAMBDA(const int i_particle) {
175  auto &particle = list._owned_data(i_particle);
176  auto &status = particle.properties.status;
177  if (status == MC::CellStatus::DEAD)
178  {
179  return;
180  }
181 
182  const size_t i_compartment = particle.properties.current_container;
183 
184  KernelInline::handle_move(i_compartment,
185  particle,
186  local_compartments,
187  neighbors,
188  local_rng,
189  _diag_transition,
190  cumulative_probability,
191  events,
192  d_t);
193 
194  const double random_number = local_rng.double_uniform();
195 
196  for (size_t i = 0; i < local_index_leaving_flow.size(); ++i)
197  {
198  const auto &index = local_index_leaving_flow(i);
199  const auto &flow = local_leaving_flow(i);
200  KernelInline::handle_exit(
201  random_number, d_t, particle, local_compartments, index, flow);
202  }
203 
204  if (status == MC::CellStatus::OUT)
205  {
206  events.template incr<MC::EventType::Exit>();
207  status = MC::CellStatus::DEAD;
208  return;
209  }
210 
211  particle.update(d_t,
212  local_compartments(i_compartment).concentrations,
213  list.rng_instance);
214  particle.contribution(contribs);
215 
216  if (status == MC::CellStatus::CYTOKINESIS)
217  {
218  events.template incr<MC::EventType::NewParticle>();
219 
220  const auto new_particle = particle.division();
221  if constexpr (!const_number_simulation)
222  {
223  auto *np = rview().extra_process.spawn();
224  if (np != nullptr)
225  {
226  *np = new_particle;
227  }
228  else
229  {

231  Kokkos::atomic_increment(&rview().waiting_allocation_particle);
232  }
233  }
234  else
235  {
236  particle.properties.weight *= 2;
237  Kokkos::atomic_increment(
238  &local_compartments(particle.properties.current_container)
239  .n_cells);
