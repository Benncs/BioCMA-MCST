struct ModelExample
{

a,
__COUNT__ // This member is recommended to automatically get the number of defined property and
// there get the right size

"example"; // Name is optional refers to model name
// Define this line if models works if unique weight, if each MCparticle has

using SelfParticle = MC::ParticlesModel<Self::n_var, Self::FloatType>; // Mandatory Type alias

init(const MC::KPRNG::pool_type& random_pool, std::size_t idx, const SelfParticle& arr);

KOKKOS_INLINE_FUNCTION static void contribution(std::size_t idx,

};
