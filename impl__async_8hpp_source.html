106 template <POD_t T>
107 int _broadcast_unsafe(T* data,
108 size_t _size,
109 size_t root,
110 MPI_Request& request)
111 {
112 if (data == nullptr)
113 {
114 throw std::invalid_argument("Data pointer is null");
115 }
116 if (_size == 0 || _size > std::numeric_limits<size_t>::max())
117 {
118 throw std::invalid_argument("Error size");
119 }
120
121 int comm_size = 0;
122 MPI_Comm_size(MPI_COMM_WORLD, &comm_size);
123 if (root >= static_cast<size_t>(comm_size))
124 {
125 throw std::invalid_argument("Root process rank is out of range");
126 }
127
128 return MPI_Ibcast(data,
129 _size,
130 get_type<T>(),
131 static_cast<int>(root),
132 MPI_COMM_WORLD,
133 &request);
134 }
