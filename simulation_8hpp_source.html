177 void SimulationUnit::cycleProcess(auto&& container, double d_t)
178 {
179 PROFILE_SECTION("cycleProcess")
180 using CurrentModel = typename std::remove_reference<decltype(container)>::type::UsedModel;
181 const size_t n_particle = container.n_particles();
182
183 auto local_rng = mc_unit->rng;
184 auto events = mc_unit->events;
185 auto contribs = get_kernel_contribution();
186
187 MC::ContributionView contribs_scatter(contribs);
188
189 this->move_info.cumulative_probability = get_kernel_cumulative_proba();
190 this->move_info.diag_transition = get_kernel_diagonal();
191 this->move_info.neighbors = mc_unit->domain.getNeighbors();
192
193 std::size_t out_total = 0;
194 std::size_t dead_total = 0;
195 std::size_t _waiting_allocation_particle = 0;
196
197 auto f = Simulation::KernelInline::CycleFunctor<CurrentModel>(
198 d_t, container, local_rng.random_pool, getkernel_concentration(), contribs_scatter, events);
199
200 _policy = MC::get_policty(f, n_particle, true);
201
202 bool enable_move = move_info.liquid_volume.size() > 1;
203 bool enable_leave = move_info.leaving_flow.size() != 0;
204
205 switch (static_cast<int>(enable_leave) * 2 + static_cast<int>(enable_move))
206 {
207 case 0:
208 // No need to move/leave
209 break;
210 case 1:
211 LAUNCH_KERNEL_MOVE(KernelInline::disable_leave, KernelInline::enable_move);
212 break;
213 case 2:
214 LAUNCH_KERNEL_MOVE(KernelInline::enable_leave, KernelInline::disable_move);
215 break;
216 case 3:
217 LAUNCH_KERNEL_MOVE(KernelInline::enable_leave, KernelInline::enable_move);
218 break;
219 }
220
221 Kokkos::fence("fence_mc_cycle_process_move");
222 Kokkos::parallel_reduce("cycle_model", _policy, f, _waiting_allocation_particle, dead_total);
223 Kokkos::fence("fence_mc_cycle_process_model");
224
225 internal_counter_dead += out_total;
226 internal_counter_dead += dead_total;
227 Kokkos::Experimental::contribute(contribs, contribs_scatter);
228
229 static constexpr uint64_t minimum_dead_particle_removal = 100;
230 const auto threshold =
231 std::max(minimum_dead_particle_removal,
232 static_cast<uint64_t>(static_cast<double>(n_particle) *
233 AutoGenerated::dead_particle_ratio_threshold));
234
235 if (internal_counter_dead > threshold)
236 {
237 container.clean_dead(internal_counter_dead);
238 internal_counter_dead = 0;
239 }
240
241 container.merge_buffer();
242
243 if (_waiting_allocation_particle != 0)
244 {
245 std::cout << "TODO: Overflow not implemented" << std::endl;
246 }
247
248 set_kernel_contribs_to_host();
249 }
