177 void SimulationUnit::cycleProcess(auto&& container, double d_t)
178 {
179 PROFILE_SECTION("cycleProcess")
180 using CurrentModel = typename std::remove_reference<decltype(container)>::type::UsedModel;
181 const size_t n_particle = container.n_particles();
182 if(n_particle==0)
183 {
184 return;
185 }
186
187 auto local_rng = mc_unit->rng;
188 auto events = mc_unit->events;
189 auto contribs = get_kernel_contribution();
190
191 MC::ContributionView contribs_scatter(contribs);
192
193 this->move_info.cumulative_probability = get_kernel_cumulative_proba();
194 this->move_info.diag_transition = get_kernel_diagonal();
195 this->move_info.neighbors = mc_unit->domain.getNeighbors();
196
197 std::size_t out_total = 0;
198 std::size_t dead_total = 0;
199 std::size_t _waiting_allocation_particle = 0;
200 auto reaction_functor = Simulation::KernelInline::CycleFunctor<CurrentModel>(
201 d_t, container, local_rng.random_pool, getkernel_concentration(), contribs_scatter, events);
202 _policy = MC::get_policy(reaction_functor, n_particle, true);
203
204 bool enable_move = move_info.liquid_volume.size() > 1;
205 bool enable_leave = move_info.leaving_flow.size() != 0;
206
207 switch (static_cast<int>(enable_leave) * 2 + static_cast<int>(enable_move))
208 {
209 case 0:
210 // No need to move/leave
211 break;
212 case 1:
213 LAUNCH_KERNEL_MOVE(KernelInline::disable_leave, KernelInline::enable_move);
214 break;
215 case 2:
216 LAUNCH_KERNEL_MOVE(KernelInline::enable_leave, KernelInline::disable_move);
217 break;
218 case 3:
219 LAUNCH_KERNEL_MOVE(KernelInline::enable_leave, KernelInline::enable_move);
220 break;
221 }
222
223 Kokkos::fence("fence_mc_cycle_process_move");
224 bool f_reaction = true; //FIXME
225 if (f_reaction)
226 {
227
228 Kokkos::parallel_reduce(
229 "cycle_model", _policy, reaction_functor, _waiting_allocation_particle, dead_total);
230 Kokkos::fence("fence_mc_cycle_process_model");
231 container.counter += out_total;
232 container.counter += dead_total;
233 Kokkos::Experimental::contribute(contribs, contribs_scatter);
234 }
235
236 const auto threshold =
237 std::max(minimum_dead_particle_removal,
238 static_cast<uint64_t>(static_cast<double>(n_particle) *
239 AutoGenerated::dead_particle_ratio_threshold));
240
241 // TODO: May change threshold as container is now cleaned before exporting,
242 if (container.counter > threshold)
243 {
244 container.clean_dead(container.counter);
245 container.counter = 0;
246 }
247 internal_counter_dead = container.counter;
248
249 container.merge_buffer();
250
251 if (_waiting_allocation_particle != 0)
252 {
253 std::cout << "TODO: Overflow not implemented" << std::endl;
254 }
255
256 set_kernel_contribs_to_host();
257 }
