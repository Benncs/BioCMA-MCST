173 void
174 SimulationUnit::cycleProcess(auto& container,
175 double d_t,
176 auto& cycle_functors)
177 {
178 PROFILE_SECTION("cycleProcess")
179 using CurrentModel =
180 typename std::remove_reference<decltype(container)>::type::UsedModel;
181 const size_t n_particle = container.n_particles();
182 if (n_particle == 0)
183 {
184 return;
185 }
186
187 pre_cycle(container, d_t, cycle_functors);
188
189 if (f_reaction)
190 {
191 this->contribs_scatter.reset();
192 cycle_functors.launch_model(n_particle);
193 }
194
195 cycle_functors.move_space.fence();
196 if (cycle_functors.move_kernel.need_launch())
197 {
198 cycle_functors.launch_move(n_particle);
199 }
200
201 post_cycle<CurrentModel>(container, cycle_functors);
202 }
203
204 template <ModelType Model>
205 void
206 SimulationUnit::post_cycle(MC::ParticlesContainer<Model>& container,
207 auto& cycle_functors)
208 {
209 PROFILE_SECTION("Simulation::post_cycle")
210 Kokkos::fence();
211 this->scatter_contribute();
212 auto [host_red, host_out_counter] = cycle_functors.get_host_reduction();
213
214 container.update_and_remove_inactive(host_out_counter, host_red.dead_total);
215
216 container.merge_buffer();
217
218 if (host_red.waiting_allocation_particle != 0)
219 {
220 if (logger)
221 {
222 logger->alert("Simulation",
223 "Overflow of particle not implemented yet (ignore "
224 "_waiting_allocation_particle)");
225 }
226 }
227
228 set_kernel_contribs_to_host();
229 }
