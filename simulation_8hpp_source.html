199 void SimulationUnit::cycleProcess(auto&& container,
200 double d_t,
201 auto& cycle_functors)
202 {
203
204 PROFILE_SECTION("cycleProcess")
205 using CurrentModel =
206 typename std::remove_reference<decltype(container)>::type::UsedModel;
207 const size_t n_particle = container.n_particles();
208 if (n_particle == 0)
209 {
210 return;
211 }
212
213 pre_cycle(container, d_t, cycle_functors);
214
215 if (cycle_functors.move_kernel.need_launch())
216 {
217 cycle_functors.launch_move(n_particle);
218 }
219 if (f_reaction)
220 {
221 cycle_functors.launch_model(n_particle);
222 }
223
224 post_cycle<CurrentModel>(container, cycle_functors);
225 }

227 template <ModelType Model>
228 void SimulationUnit::post_cycle(MC::ParticlesContainer<Model>& container,
229 auto& cycle_functors)
230 {
231 Kokkos::fence();
232 this->scatter_contribute();
233 auto [host_red, host_out_counter] = cycle_functors.get_host_reduction();
234
235 container.update_and_remove_inactive(host_out_counter, host_red.dead_total);
236
237 container.merge_buffer();
238
239 if (host_red.waiting_allocation_particle != 0)
240 {
241 if (logger)
242 {
243 logger->alert("Simulation",
244 "Overflow of particle not implemented yet (ignore "
245 "_waiting_allocation_particle)");
246 }
247 }
248
249 set_kernel_contribs_to_host();
250 }
