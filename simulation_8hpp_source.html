183 void
184 SimulationUnit::cycleProcess(auto& container,
185 double d_t,
186 auto& cycle_functors)
187 {
188 PROFILE_SECTION("cycleProcess")
189 using CurrentModel =
190 typename std::remove_reference<decltype(container)>::type::UsedModel;
191 const size_t n_particle = container.n_particles();
192 if (n_particle == 0)
193 {
194 return;
195 }
196
197 pre_cycle(container, d_t, cycle_functors);
198
199 if (f_reaction)
200 {
201 this->contribs_scatter.reset();
202 cycle_functors.launch_model(n_particle);
203 }
204
205 if (cycle_functors.move_kernel.need_launch())
206 {
207 cycle_functors.launch_move(n_particle);
208 }
209
210 post_cycle<CurrentModel>(container, cycle_functors);
211 }

213 template <ModelType Model>
214 void
215 SimulationUnit::post_cycle(MC::ParticlesContainer<Model>& container,
216 auto& cycle_functors)
217 {
218 PROFILE_SECTION("Simulation::post_cycle")
219 Kokkos::fence();
220 this->scatter_contribute();
221 const auto [host_red, host_out_counter]
222 = cycle_functors.get_host_reduction();
223
224 container.update_and_remove_inactive(host_out_counter, host_red.dead_total);
225
226 container.merge_buffer();
227
228 if (host_red.waiting_allocation_particle != 0)
229 {
230 if (logger)
231 {
232 logger->alert("Simulation",
233 "Overflow of particle not implemented yet (ignore "
234 "_waiting_allocation_particle)");
235 }
236 }
