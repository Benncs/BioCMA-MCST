131 KOKKOS_INLINE_FUNCTION void
132 TwoMetaDiv::init([[maybe_unused]] const MC::KPRNG::pool_type& random_pool,
133 std::size_t idx,
134 const SelfParticle& arr)
135 {
136 constexpr auto local_lc = length_c_dist;
137 constexpr auto length_dist =
138 MC::Distributions::TruncatedNormal<FloatType>(l_c_m / 2, l_c_m / 5., l_min_m, l_max_m);
139
140 constexpr auto mu_nu_dist = nu_max_kg_s * 0.1;
141 constexpr auto nu_1_initial_dist = MC::Distributions::TruncatedNormal<float>(
142 mu_nu_dist, mu_nu_dist / 7., 0., static_cast<double>(nu_max_kg_s));
143
144 auto gen = random_pool.get_state();
145 GET_PROPERTY(Self::particle_var::length) = length_dist.draw(gen);
146 GET_PROPERTY(Self::particle_var::l_cp) = local_lc.draw(gen);
147 GET_PROPERTY(particle_var::nu1) = nu_1_initial_dist.draw(gen);
148 random_pool.free_state(gen);
149 GET_PROPERTY(particle_var::contrib_phi_s) = 0;
150 Uptake<Self>::init(random_pool, idx, arr);
151 }

153 KOKKOS_INLINE_FUNCTION MC::Status TwoMetaDiv::update(const MC::KPRNG::pool_type& random_pool,
154 FloatType d_t,
155 std::size_t idx,
156 const SelfParticle& arr,
157 const MC::LocalConcentration& concentrations)
158 {
159 (void)random_pool;
160 const auto phi_s =
161 Uptake<Self>::uptake_step(phi_s_max, phi_perm_max, d_t, idx, arr, concentrations);
162
163 const auto o = Kokkos::max(static_cast<float>(concentrations(1)), 0.F);
164
165 const float phi_o2 = (phi_o2_max)*o / (o + k_o); // gO2/s
166
167 const float nu_1_star = y_sx_1 * MolarMassG *
168 Kokkos::min(phi_s / MolarMassG,
169 phi_o2 / MolarMassO2 / y_os_molar); // gX/s
170
171 const float s_1_star = (1 / y_sx_1 * nu_1_star);
172
173 const float phi_s_residual_1_star = Kokkos::max(phi_s - s_1_star, 0.F);
174 KOKKOS_ASSERT(phi_s_residual_1_star >= 0.F);
175
176 const float nu_2_star = y_sx_2 * phi_s_residual_1_star; // gX/s
177
178 GET_PROPERTY(Self::particle_var::nu_eff_1) =
179 Kokkos::min(nu_1_star, GET_PROPERTY(Self::particle_var::nu1)); // gX/s
180
181 const float s_1 = (1 / y_sx_1 * GET_PROPERTY(Self::particle_var::nu_eff_1));
182 const float phi_s_residual_1 = Kokkos::max(phi_s - s_1, 0.F);
183 GET_PROPERTY(Self::particle_var::nu_eff_2) =
184 Kokkos::min(y_sx_2 * phi_s_residual_1, GET_PROPERTY(Self::particle_var::nu2)); // gX/s
185
186 const float s_growth = s_1 + (1 / y_sx_2 * GET_PROPERTY(Self::particle_var::nu_eff_2));
187
188 const float s_overflow = phi_s - s_growth;
189
190 KOKKOS_ASSERT(GET_PROPERTY(Self::particle_var::nu_eff_1) >= 0.F);
191 KOKKOS_ASSERT(GET_PROPERTY(Self::particle_var::nu_eff_2) >= 0.F);
192
193 // CONTRIBS
194 GET_PROPERTY(Self::particle_var::contrib_phi_s) = -phi_s;
195 GET_PROPERTY(Self::particle_var::contrib_phi_o2) =
196 -1 * ((1. / y_sx_1 / MolarMassG * y_os_molar * MolarMassO2 *
197 GET_PROPERTY(Self::particle_var::nu_eff_1)) +
198 0. * GET_PROPERTY(Self::particle_var::nu_eff_2));
199
200 GET_PROPERTY(Self::particle_var::contrib_phi_ac) =
201 GET_PROPERTY(Self::particle_var::nu_eff_2) / y_sx_2 * y_sa +
202 (s_overflow > 0. ? y_sa * (s_overflow) : 0);
203
204 // ODE
205 GET_PROPERTY(Self::particle_var::nu1) +=
206 static_cast<float>(d_t) * ((nu_1_star - GET_PROPERTY(Self::particle_var::nu1)) / tau_1);
207
208 GET_PROPERTY(Self::particle_var::nu2) +=
209 static_cast<float>(d_t) * ((nu_2_star - GET_PROPERTY(Self::particle_var::nu2)) / tau_2);
210
211 const auto sum_nu =
212 (GET_PROPERTY(Self::particle_var::nu_eff_1) + GET_PROPERTY(Self::particle_var::nu_eff_2));
213
214 GET_PROPERTY(Self::particle_var::length) += static_cast<float>(d_t) * (sum_nu / lin_density);
215
216 GET_PROPERTY(Self::particle_var::age) += d_t;
217
218 return (GET_PROPERTY(Self::particle_var::length) > GET_PROPERTY(Self::particle_var::l_cp))
219 ? MC::Status::Division
220 : MC::Status::Idle;
221 }

223 KOKKOS_INLINE_FUNCTION void TwoMetaDiv::division(const MC::KPRNG::pool_type& random_pool,
224 std::size_t idx,
225 std::size_t idx2,
226 const SelfParticle& arr,
227 const SelfParticle& child_buffer_arr)
228 {
229 constexpr auto local_lc = length_c_dist;
230 const FloatType new_current_length =
231 GET_PROPERTY(particle_var::length) / static_cast<FloatType>(2.);
232
233 GET_PROPERTY(Self::particle_var::length) = new_current_length;
234 GET_PROPERTY(Self::particle_var::age) = 0;
235
236 GET_PROPERTY_FROM(idx2, child_buffer_arr, Self::particle_var::length) = new_current_length;
237 GET_PROPERTY_FROM(idx2, child_buffer_arr, Self::particle_var::age) = 0;
238
239 auto nu_1_o = GET_PROPERTY_FROM(idx, arr, Self::particle_var::nu_eff_1);
240 auto nu_2_o = GET_PROPERTY_FROM(idx, arr, Self::particle_var::nu_eff_2);
241 auto gen = random_pool.get_state();
242 GET_PROPERTY(Self::particle_var::l_cp) = local_lc.draw(gen);
243 if (nu_1_o != 0)
244 {
245 GET_PROPERTY_FROM(idx2, child_buffer_arr, Self::particle_var::nu1) =
246 MC::Distributions::TruncatedNormal<FloatType>::draw_from(gen, nu_1_o, nu_1_o / 2., 0, 1.);
247 }
248 if (nu_2_o != 0)
249 {
250 GET_PROPERTY_FROM(idx2, child_buffer_arr, Self::particle_var::nu2) =
251 MC::Distributions::TruncatedNormal<FloatType>::draw_from(gen, nu_2_o, nu_2_o / 2., 0, 1.);
252 }
253
254 GET_PROPERTY_FROM(idx2, child_buffer_arr, Self::particle_var::l_cp) = local_lc.draw(gen);
255 random_pool.free_state(gen);
256
257 Uptake<Self>::division(random_pool, idx, idx2, arr, child_buffer_arr);
258 }
