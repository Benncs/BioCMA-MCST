20concat_arrays(const std::array<std::string_view, N1>& arr1,
21 const std::array<std::string_view, N2>& arr2)
22{
23 std::array<std::string_view, N1 + N2> result;
24 std::copy(arr1.begin(), arr1.end(), result.begin());
