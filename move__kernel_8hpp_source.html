43 KOKKOS_INLINE_FUNCTION std::size_t
44 __find_next_compartment(const ConstNeighborsView<ComputeSpace>& neighbors,
45 const CumulativeProbabilityView<ComputeSpace>& cumulative_probability,
46 const std::size_t i_compartment,
47 const double random_number)
48 {
49 const int max_neighbor = static_cast<int>(neighbors.extent(1));
50
51 std::size_t next = neighbors(i_compartment, 0); // Default to the first neighbor
52
53 // Iterate through the neighbors to find the appropriate next compartment
54 for (int k_neighbor = 0; k_neighbor < max_neighbor - 1; ++k_neighbor)
55 {
56
57 // Get the cumulative probability range for the current neighbor
58 const auto pi = cumulative_probability(i_compartment, k_neighbor);
59 const auto pn = cumulative_probability(i_compartment, k_neighbor + 1);
60
61 // Use of a Condition mask to avoid branching.
62 next = (random_number <= pn && pi <= random_number) ? neighbors(i_compartment, k_neighbor + 1)
63 : next;
64 }
65
66 return next; // Return the index of the chosen next compartment
67 }

108 KOKKOS_FUNCTION void handle_move(const std::size_t idx) const
109 {
110 auto generator = random_pool.get_state();
111 const float rng1 = generator.frand(0., 1.);
112 const double rng2 = generator.drand(0., 1.);
113 random_pool.free_state(generator);
114
115 const std::size_t i_current_compartment = positions(idx);
116
117 const bool mask_next = probability_leaving(rng1,
118 move.liquid_volume(i_current_compartment),
119 move.diag_transition(i_current_compartment),
120 d_t);
121
122 positions(idx) =
123 (mask_next)
124 ? __find_next_compartment(
125 move.neighbors, move.cumulative_probability, i_current_compartment, rng2)

147 random_pool.free_state(generator);
148
