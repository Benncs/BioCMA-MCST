29 KOKKOS_INLINE_FUNCTION std::size_t __find_next_compartment(
30 const ConstNeighborsView<ComputeSpace>& neighbors,
31 const CumulativeProbabilityView<ComputeSpace>& cumulative_probability,
32 const std::size_t i_compartment,
33 const double random_number)
34 {
35 const int max_neighbor = static_cast<int>(neighbors.extent(1));
36
37 std::size_t next =
38 neighbors(i_compartment, 0); // Default to the first neighbor
39
40 // Iterate through the neighbors to find the appropriate next compartment
41 for (int k_neighbor = 0; k_neighbor < max_neighbor - 1; ++k_neighbor)
42 {
43
44 // Get the cumulative probability range for the current neighbor
45 const auto pi = cumulative_probability(i_compartment, k_neighbor);
46 const auto pn = cumulative_probability(i_compartment, k_neighbor + 1);
47
48 // Use of a Condition mask to avoid branching.
49 next = (random_number <= pn && pi <= random_number)
50 ? neighbors(i_compartment, k_neighbor + 1)
51 : next;
52 }
53
54 return next; // Return the index of the chosen next compartment
55 }

237 KOKKOS_FUNCTION void handle_move(const std::size_t idx) const
238 {
239 auto generator = random_pool.get_state();
240 const float rng1 = generator.frand(0., 1.);
241 const double rng2 = generator.drand(0., 1.);
242 random_pool.free_state(generator);
243
244 const std::size_t i_current_compartment = positions(idx);
245
246 KOKKOS_ASSERT(i_current_compartment < move.liquid_volume.extent(0));
247
248 const bool mask_next =
249 probability_leaving<int>(rng1,
250 move.liquid_volume(i_current_compartment),
251 move.diag_transition(i_current_compartment),
252 d_t);
253
254 positions(idx) =
255 (mask_next) ? __find_next_compartment(move.neighbors,
256 move.cumulative_probability,
257 i_current_compartment,
258 rng2)
259 : i_current_compartment;
260
261 KOKKOS_ASSERT(positions(idx) < move.liquid_volume.extent(0));
262
263 if constexpr (AutoGenerated::FlagCompileTime::enable_event_counter)
264 {
265 if (mask_next)
266 {
267 events.wrap_incr<MC::EventType::Move>();
268 }
269 }

272 KOKKOS_INLINE_FUNCTION void handle_exit(std::size_t idx,
273 std::size_t& dead_count) const
274 {
275 const auto position = positions(idx);
276 const std::size_t n_flow = move.leaving_flow.size();
277 const auto liquid_volume = move.liquid_volume(position);
278
279 for (std::size_t i = 0LU; i < n_flow; ++i)
280 {
281
282 // const float random_number = random(idx, i);
283 auto generator = random_pool.get_state();
284 const float random_number = generator.frand(0., 1.);
285 random_pool.free_state(generator);
286
287 const auto& [index, flow] = move.leaving_flow(i);
288
289 const bool is_leaving =
290 (position == index) &&
291 probability_leaving(random_number, liquid_volume, flow, d_t);
292
293 const int leave_mask = static_cast<int>(is_leaving);
294
295 // If using probes
296 if constexpr (AutoGenerated::FlagCompileTime::use_probe)
297 {
298 // Execute probe set, but only actually do something if leaving
299 const auto probe_value = ages(idx, 0);
300 if (is_leaving && !probes.set(probe_value))
301 {
302 Kokkos::printf("[Kernel]: PROBES OVERFLOW\r\n");
303 }
304 }
305 if constexpr (AutoGenerated::FlagCompileTime::enable_event_counter)
306 {
307 events.add<MC::EventType::Exit>(leave_mask);
308 }
309
310 dead_count += leave_mask;
311 ages(idx, 0) = leave_mask * 0 + (1 - leave_mask) * ages(idx, 0);
312 status(idx) = is_leaving ? MC::Status::Exit : status(idx);
313
314 // events.wrap_incr<MC::EventType::Exit>();
