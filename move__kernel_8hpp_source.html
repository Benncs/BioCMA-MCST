45 KOKKOS_INLINE_FUNCTION std::size_t __find_next_compartment(
46 const ConstNeighborsView<ComputeSpace>& neighbors,
47 const CumulativeProbabilityView<ComputeSpace>& cumulative_probability,
48 const std::size_t i_compartment,
49 const double random_number)
50 {
51 const int max_neighbor = static_cast<int>(neighbors.extent(1));
52
53 std::size_t next =
54 neighbors(i_compartment, 0); // Default to the first neighbor
55
56 // Iterate through the neighbors to find the appropriate next compartment
57 for (int k_neighbor = 0; k_neighbor < max_neighbor - 1; ++k_neighbor)
58 {
59
60 // Get the cumulative probability range for the current neighbor
61 const auto pi = cumulative_probability(i_compartment, k_neighbor);
62 const auto pn = cumulative_probability(i_compartment, k_neighbor + 1);
63
64 // Use of a Condition mask to avoid branching.
65 next = (random_number <= pn && pi <= random_number)
66 ? neighbors(i_compartment, k_neighbor + 1)
67 : next;
68 }
69
70 return next; // Return the index of the chosen next compartment
71 }

150 KOKKOS_FUNCTION void handle_move(const std::size_t idx) const
151 {
152 auto generator = random_pool.get_state();
153 const float rng1 = generator.frand(0., 1.);
154 const double rng2 = generator.drand(0., 1.);
155 random_pool.free_state(generator);
156
157 const std::size_t i_current_compartment = positions(idx);
158
159 KOKKOS_ASSERT(i_current_compartment < move.liquid_volume.extent(0));
160
161 const bool mask_next =
162 probability_leaving(rng1,
163 move.liquid_volume(i_current_compartment),
164 move.diag_transition(i_current_compartment),
165 d_t);
166
167 positions(idx) =
168 (mask_next) ? __find_next_compartment(move.neighbors,
169 move.cumulative_probability,
170 i_current_compartment,
171 rng2)
172 : i_current_compartment;
173
174 KOKKOS_ASSERT(positions(idx) < move.liquid_volume.extent(0));
175
176 if constexpr (AutoGenerated::FlagCompileTime::enable_event_counter)
177 {
178 if (mask_next)
179 {
180 events.wrap_incr<MC::EventType::Move>();
181 }
182 }
183 }

185 KOKKOS_FUNCTION void handle_exit(std::size_t idx,
186 std::size_t& dead_count) const
187 {
188
189 for (size_t i = 0LU; i < move.index_leaving_flow.size(); ++i)
190 {
191 const auto position = positions(idx);
192 auto generator = random_pool.get_state();
193 const float random_number = generator.frand(0., 1.);
194 random_pool.free_state(generator);
195
196 const auto& index = move.index_leaving_flow(i);
197 const auto& flow = move.leaving_flow(i);
198 if (position != index)
199 {
200 return;
201 }
202 if (probability_leaving(
203 random_number, move.liquid_volume(position), flow, d_t))
204 {
205 dead_count += 1;
206 status(idx) = MC::Status::Exit;
207 events.wrap_incr<MC::EventType::Exit>();
208 if constexpr (AutoGenerated::FlagCompileTime::use_probe)
209 {
210 // Ignore ret value, if probe is full were gonna miss events which
211 // is not really important
212 if (!probes.set(ages(idx, 0)))
213 {
214 Kokkos::printf("[Kernel]: PROBES OVERFLOW\r\n");
215 };
216 ages(idx, 0) = 0;
217 }
218 }
219 }
