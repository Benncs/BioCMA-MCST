28 template <HasExportPropertiesFull Model, typename ExecutionSpace>
29 void inner(std::size_t n_p,
30 const MC::ParticlePositions& position,
31 const typename Model::SelfParticle& model,
32 const MC::ParticleAges& ages,
33 ParticlePropertyViewType<ComputeSpace>& particle_values,
34 ParticlePropertyViewType<ComputeSpace>& spatial_values,
35 ParticlePropertyViewType<ComputeSpace>& ages_value,
36 const MC::ParticleStatus& status)
37 {
38 Kokkos::Experimental::
39 ScatterView<double**, Kokkos::LayoutRight, ComputeSpace>
40 scatter_spatial_values(spatial_values);
41
42 Kokkos::parallel_for(
43 "kernel_get_properties",
44 Kokkos::RangePolicy<ExecutionSpace>(0, n_p),
45 KOKKOS_LAMBDA(const int i_particle) {
46 if (status(i_particle) != MC::Status::Idle)
47 {
48 return;
49 }
50 auto access = scatter_spatial_values.access();
51
52 for (std::size_t i = 0; i < Model::n_var; ++i)
53 {
54 access(i, position(i_particle)) += model(i_particle, i);
55 particle_values(i, i_particle) = model(i_particle, i);
56 }
57 auto mass = Model::mass(i_particle, model);
58 access(Model::n_var, position(i_particle)) += mass;
59 particle_values(Model::n_var, i_particle) = mass;
60 ages_value(0, i_particle) = ages(i_particle, 0);
61 ages_value(1, i_particle) = ages(i_particle, 1);
62 });
63 Kokkos::fence();
64 Kokkos::Experimental::contribute(spatial_values, scatter_spatial_values);
65 }

67 template <HasExportPropertiesPartial Model, typename ExecutionSpace>
68 void inner_partial(const std::size_t n_p,
69 const MC::ParticlePositions& position,
70 const typename Model::SelfParticle& model,
71 const MC::ParticleAges& ages,
72 ParticlePropertyViewType<ComputeSpace>& particle_values,
73 ParticlePropertyViewType<ComputeSpace>& spatial_values,
74 ParticlePropertyViewType<ComputeSpace>& ages_value,
75 const MC::ParticleStatus& status)
76 {
77 Kokkos::Experimental::
78 ScatterView<double**, Kokkos::LayoutRight, ComputeSpace>
79 scatter_spatial_values(spatial_values);
80 static const auto indices = Model::get_number();
81
82 static Kokkos::View<size_t*, HostSpace> host_index("host_index",
83 indices.size());
84 for (std::size_t i = 0; i < indices.size(); ++i)
85 {
86 host_index(i) = indices[i];
87 }
88
89 const Kokkos::View<const size_t*, ComputeSpace> kindices =
90 Kokkos::create_mirror_view_and_copy(ComputeSpace(), host_index);
91 Kokkos::parallel_for(
92 "kernel_get_properties",
93 Kokkos::RangePolicy<ExecutionSpace>(0, n_p),
94 KOKKOS_LAMBDA(const int i_particle) {
95 if (status(i_particle) != MC::Status::Idle)
96 {
97 return;
98 }
99
100 auto access = scatter_spatial_values.access();
101
102 for (size_t k_var = 0; k_var < kindices.extent(0); ++k_var)
103 {
104 const auto index_export = kindices(k_var);
105 const auto current = model(i_particle, index_export);
106
107 access(k_var, position(i_particle)) += current;
108 particle_values(k_var, i_particle) = current;
109 }
110
111 const auto mass = Model::mass(i_particle, model);
112 access(kindices.extent(0), position(i_particle)) += mass;
113 particle_values(kindices.extent(0), i_particle) = mass;
114 ages_value(0, i_particle) = ages(i_particle, 0);
115 ages_value(1, i_particle) = ages(i_particle, 1);
116 });
117 Kokkos::fence();
118 Kokkos::Experimental::contribute(spatial_values, scatter_spatial_values);
119 }

121 template <ModelType M>
122 std::optional<PostProcessing::BonceBuffer>
123 get_properties(MC::ParticlesContainer<M>& container,
124 const std::size_t n_compartment,
125 const bool with_age)
126 {
127 if constexpr (HasExportProperties<M>)
128 {
129 container.force_remove_dead();
130 BonceBuffer properties;
131 const std::size_t n_p =
132 container.n_particles(); // USE list size not Kokkos View size.
133 // bcause container allocates more particles
134 // than needed
135 auto ar = M::names();
136 properties.vnames = std::vector<std::string>(ar.begin(), ar.end());
137 properties.vnames.emplace_back("mass");
138
139 std::size_t n_var = M::n_var + 1;
140 if constexpr (HasExportPropertiesPartial<M>)
141 {
142 n_var = ar.size() + 1;
143 }
144
145 ParticlePropertyViewType<ComputeSpace> spatial_values(
146 "property_spatial", n_var, n_compartment);
147 ParticlePropertyViewType<ComputeSpace> particle_values(
148 "property_values", n_var, n_p);
149
150 // TODO Currently view is allocated and filled even if with_age flag is
151 // turned of, with_age controlls export but not postprocessing
152 ParticlePropertyViewType<ComputeSpace> ages_values("ages_values", 2, n_p);
153
154 if constexpr (HasExportPropertiesPartial<M>)
155 {
156 inner_partial<M, Kokkos::DefaultExecutionSpace>(n_p,
157 container.position,
158 container.model,
159 container.ages,
160 particle_values,
161 spatial_values,
162 ages_values,
163 container.status);
164 }
165 else
166 {
167
168 inner<M, Kokkos::DefaultExecutionSpace>(n_p,
169 container.position,
170 container.model,
171 container.ages,
172 particle_values,
173 spatial_values,
174 ages_values,
175 container.status);
176 }
177
178 properties.particle_values = Kokkos::create_mirror_view_and_copy(
179 Kokkos::HostSpace(), particle_values);
180 properties.spatial_values = Kokkos::create_mirror_view_and_copy(
181 Kokkos::HostSpace(), spatial_values);
182
183 if (with_age)
184 {
185
186 properties.ages = Kokkos::create_mirror_view_and_copy(
187 Kokkos::HostSpace(), ages_values);
188 }
189 else
190 {
191 properties.ages = std::nullopt;
192 }
193
194 return properties;
195 }
196 else
197 {
198 return std::nullopt;
199 }
200 }
